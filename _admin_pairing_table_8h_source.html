162  ConstAdminIterator(const AdminPairingInfo * start, size_t index, size_t maxSize) :
163  mStart(start), mIndex(index), mMaxSize(maxSize)
164  {
165  if (mIndex >= maxSize)
166  {
167  mIndex = maxSize;
168  }
169  else if (!mStart[mIndex].IsInitialized())
170  {
171  Advance();
172  }
173  }
174  ConstAdminIterator(const ConstAdminIterator &) = default;
175  ConstAdminIterator & operator=(const ConstAdminIterator &) = default;
176 
177  ConstAdminIterator & operator++() { return Advance(); }
178  ConstAdminIterator operator++(int)
179  {
180  ConstAdminIterator other(*this);
181  Advance();
182  return other;
183  }
184 
185  const AdminPairingInfo & operator*() const { return mStart[mIndex]; }
186  const AdminPairingInfo * operator->() const { return mStart + mIndex; }
187 
188  bool operator==(const ConstAdminIterator & other)
189  {
190  if (IsAtEnd())
191  {
192  return other.IsAtEnd();
193  }
194 
195  return (mStart == other.mStart) && (mIndex == other.mIndex) && (mMaxSize == other.mMaxSize);
196  }
197  bool operator!=(const ConstAdminIterator & other) { return !(*this == other); }
198 
199  bool IsAtEnd() const { return (mIndex == mMaxSize); }
200 
201 private:
202  const AdminPairingInfo * mStart;
203  size_t mIndex;
204  size_t mMaxSize;
205 
206  ConstAdminIterator & Advance()
207  {
208  do
209  {
210  if (mIndex < mMaxSize)
211  {
212  mIndex++;
213  }
214  } while (!IsAtEnd() && !mStart[mIndex].IsInitialized());
215 
216  return *this;
217  }
